```bush 
git status статус синхронизации с гит 
git add . добавление индекса во всей pwd
git commit -m 'zaebalo' создание коммита с сообщением
git commit --amend позднее добавление файла в коммит
git push выгрузка в глобальный репозиторий 
git diff --staged просмотр изменения 
git restore README.md восстановление удалленого/изменнного файл
git rm удаление и подготовка к коммиту
git log просмотр истории коммитов
git log -- <имя файла>  просмотр коммитов где учавствовал файл
git log -p   с полным дифом  Мотать вперед f, мотать назад b
git show 5120bea... просмотр по хешу сокращено до 7 символов
git blame INFO.md просмотр по истории изменения файла

git grep line ищет совпадения по всем файлам проекта
git grep -i hexlet  Флаг i позволяет искать без учета регистра
git grep Hexlet 5120bea  Поиск в конкретном коммите
git grep Hexlet $(git rev-list --all)  Поиск по всей истории
 rev-list возвращает список хешей коммитов


git clean -fd  force/directory удаление всех директорий что нет в гит 
git restore --staged INFO.md удаление индекса


git revert aa600a... отменяет любой коммит из истории проекта 

git reset отмена коммита 
git reset --hard (полное удаление) HEAD~ (последний коммит)


git log --oneline  Показывает сокращенный вывод сгружая историю
git checkout <хеш коммита> Выгрузка состояние репозитория на момент нужного коммита
git checkout main выгрузка истории ветки? 
git switch - возврат назад после chekout
git branch текущий загруженный коммит 
```
